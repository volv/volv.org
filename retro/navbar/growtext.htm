Be sure to check the colour coding

  <BODY onLoad="change(); grow()">

  <SCRIPT LANGUAGE=JAVASCRIPT>
  //Created by Steven Bigham
  var size = 20;
  function grow() {
  size += 1;
  if (size <= 55) {
  growtext.style.fontSize = size
  setTimeout("grow()", 10)
  }
  else {
  shrink()
  }
  }
  function shrink() {
  size -= 1;
  if (size >= 20) {
  growtext.style.fontSize = size
  setTimeout("shrink()", 10)
  }
  else {
  grow()
  }
  }
  function change() {
  var col = new Array("blue","red","green","black","pink","purple","orange","cyan")
  var ran = Math.round(Math.random()*7)
  growtext.style.color = col[ran]
  setTimeout("change()", 500)
  }
  </SCRIPT>
